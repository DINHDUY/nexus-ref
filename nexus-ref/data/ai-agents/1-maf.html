Section 06

Middleware

⚙️Chat Middleware (Request/Response)
from agent_framework import ChatMiddleware, ChatContext

class LoggingMiddleware(ChatMiddleware):
    """Logs every request + response to/from LLM."""

    async def on_invoke(self, ctx: ChatContext, next) -> str:
        print(f"→ Request: {ctx.messages[-1].content[:80]}")
        response = await next(ctx)          # call next handler
        print(f"← Response: {str(response)[:80]}")
        return response

class RateLimitMiddleware(ChatMiddleware):
    async def on_invoke(self, ctx: ChatContext, next):
        await self.throttle()
        return await next(ctx)

agent = OpenAIResponsesClient().as_agent(
    name="MiddlewareAgent",
    instructions="You are helpful.",
    middleware=[LoggingMiddleware(), RateLimitMiddleware()],
)PYTHON
🔧Function (Tool) Middleware
from agent_framework import FunctionMiddleware, FunctionContext

class AuditToolMiddleware(FunctionMiddleware):
    """Audits every tool call before execution."""

    async def on_invoke(self, ctx: FunctionContext, next):
        print(f"TOOL CALL: {ctx.function_name}({ctx.arguments})")
        result = await next(ctx)              # execute tool
        print(f"TOOL RESULT: {str(result)[:100]}")
        return result

class ApprovalMiddleware(FunctionMiddleware):
    """Human-in-the-loop: approve dangerous tools."""

    async def on_invoke(self, ctx: FunctionContext, next):
        if "delete" in ctx.function_name:
            confirm = input(f"Approve {ctx.function_name}? [y/N]: ")
            if confirm.lower() != "y":
                return "User declined."
        return await next(ctx)PYTHON
🔷Middleware — .NET
// using Microsoft.Agents.AI;

var agent = chatClient
    .AsAIAgent(instructions: "You are helpful.")
    .AsBuilder()
    .UseOpenTelemetry(sourceName: "MyApp")   // built-in OTel
    .Use(async (ctx, next) =>               // inline middleware
    {
        Console.WriteLine($"→ {ctx.Messages.Last().Text}");
        var result = await next(ctx);
        Console.WriteLine($"← {result}");
        return result;
    })
    .Build();

// Exception handling middleware
.Use(async (ctx, next) =>
{
    try { return await next(ctx); }
    catch (Exception ex)
    { Console.Error.WriteLine(ex); return "Error occurred."; }
})C#

🔀Conditional Edges
from agent_framework import WorkflowBuilder, ConditionalEdge
from pydantic import BaseModel

class RouteDecision(BaseModel):
    route: str  # "billing" | "support" | "sales"

def router(msg) -> RouteDecision:
    # Use LLM structured output for reliable routing
    return RouteDecision(route="billing")

builder = WorkflowBuilder()
builder.add_conditional_edge(
    source=router,
    path_map={
        "billing":  billing_agent,
        "support":  support_agent,
        "sales":    sales_agent,
    },
    condition=lambda result: result.route,
)
builder.set_start_executor(router)
workflow = builder.build()PYTHON

⚙️Durable Task (Long-Running)
from agent_framework.integrations.durable_task import (
    DurableWorkflowRunner,
    checkpoint_enabled,
)

@checkpoint_enabled
async def long_workflow(ctx, input_data):
    result1 = await ctx.call_activity(step_one, input_data)
    # State saved here — survives restarts/crashes
    result2 = await ctx.call_activity(step_two, result1)
    return result2

runner = DurableWorkflowRunner()
instance_id = await runner.start(long_workflow, input="start")
status = await runner.get_status(instance_id)PYTHON
Durable Task = Workflow state persisted to Azure Storage. Survives crashes and restarts.

# FastAPI example
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from agent_framework.openai import OpenAIResponsesClient

app = FastAPI()
agent = OpenAIResponsesClient().as_agent(
    name="WebAgent", instructions="You are a web agent."
)

@app.post("/chat")
async def chat(body: dict):
    result = await agent.run(body["message"])
    return {"response": result}

@app.post("/stream")
async def stream(body: dict):
    async def gen():
        async for chunk in agent.run_streaming(body["message"]):
            yield chunk
    return StreamingResponse(gen(), media_type="text/plain")PYTHON
